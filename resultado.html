function getFullName(firstName, lastName) {

   return `${firstName} ${lastName}`;

 }

 const firstName = "John";

 const lastName = "Doe";

 const fullName = `My name is ${getFullName(firstName, lastName)}.`;

 console.log(fullName); // output: "My name is John Doe."
